FROM node:18-alpine

WORKDIR /app

# Копируем ТОЛЬКО package.json сначала для кэширования
COPY server/package.json ./
COPY server/package-lock.json* ./

# Устанавливаем зависимости
RUN npm install

# Копируем весь исходный код сервера
COPY server/ ./

# Создаем папку для клиента если нужно
RUN mkdir -p public

# Открываем порт
EXPOSE 3000

# Запускаем сервер
CMD ["node", "index.js"]
